#plot clusters
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

def plot_clusters(data_clusters):
    fig = plt.figure()
    ax = fig.add_subplot(111, projection='3d')
    cmap=plt.cm.get_cmap("Accent_r")
    
    cont=0
    for data in data_clusters:
        x = []
        y = []
        z = []
        for data_point in data:
            x.append(data_point[0])
            y.append(data_point[1])
            z.append(data_point[2])
        ax.scatter(x, y, z, c=cmap(cont), marker='o',label="cluster "+str(cont))
        cont=cont+1

    ax.set_xlabel('X Label')
    ax.set_ylabel('Y Label')
    ax.set_zlabel('Z Label')
    ax.legend()
    
    return fig

#plot lanes clusters

def get_pos(lane,role):
    pos=None
    if(role=="DUO_SUPPORT"):
        pos="sup"
    elif(role=="DUO_CARRY"):
        pos="adc"
    if(lane=="JUNGLE"):
        pos="jg"
    elif(lane=="TOP"):
        pos="top"
    elif(lane=="MIDDLE"):
        pos="mid"

    return pos

data_clusters=[[] for cont in range(6)]
dataset_labels_lane=[]

for lane,role in zip(dataset_principal.get("lane"),dataset_principal.get("role")):
    pos=get_pos(lane,role)
    label=0
    if(pos=="sup"):
        label=1
    elif(pos=="adc"):
        label=2
    elif(pos=="jg"):
        label=3
    elif(pos=="top"):
        label=4
    elif(pos=="mid"):
        label=5
    dataset_labels_lane.append(label)
    

for point,label in zip(pca_data_transform,dataset_labels_lane):
    data_clusters[label].append(point)

plot_clusters(data_clusters[1:])
