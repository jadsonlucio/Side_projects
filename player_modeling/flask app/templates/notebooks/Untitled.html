def select_lane(array):
    lane=array[0]
    role=array[1]
    pos=get_pos(lane,role)
    label=0
    if(pos=="sup"):
        label=1
    elif(pos=="adc"):
        label=2
    elif(pos=="jg"):
        label=3
    elif(pos=="top"):
        label=4
    elif(pos=="mid"):
        label=5
    return label

cluster = test.get_cluster_by_features(test.Dataset_numerico.columns,["lane","role"], select_lane)
test.Cluster=cluster
lane_cluster_transform,lane_cluster_centroids=test.get_clusters_transformed()
draw_clusters(lane_cluster_transform[2:],labels=["adc","jg","top","mid"])
